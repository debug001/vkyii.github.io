# pyenv
	brew install pyenv
	echo '# pyenv' >> ~/.bashrc
	echo 'export PYENV_ROOT=/usr/local/opt/pyenv' >> ~/.bashrc
	echo 'if which pyenv > /dev/null; then eval "$(pyenv init -)"; fi' >> ~/.bashrc
	source ~/.bashrc

# python
	pyenv install 2.7.10
	pyenv install 3.4.3
	pyenv global 2.7.10
	pyenv rehash #切换版本之后rehash使path等配置生效

# ipython
	pip install ipython
